var createError = require('http-errors');
var express = require('express');
var path = require('path');
var cookieParser = require('cookie-parser');
var logger = require('morgan');
var validator = require('express-validator');
var cors= require('cors')
 
// For Service Registery
require('dotenv').config(); 
 
// Common Routers 
var indexRouter = require('./routes/index');
// var usersRouter = require('./routes/users');
 
//TODO Better to expose Separate API Gateways and remove subfolder
 
var vehiclesRouter = require('./routes/vehicles/vehicle-routes')
var customersRouter = require('./routes/customers/customer-routes') 
 var uiRouter = require('./routes/ui-routes/ui-routes')
 
var app = express();
 
app.use(logger('dev'));
app.use(express.json());
app.use(express.urlencoded({ extended: false }));
app.use(cookieParser());
app.use(validator())
app.use(express.static(path.join(__dirname, 'public')));
app.use(cors())
 
//API Orchestration
app.use('/api/vehicles/v1', vehiclesRouter);
app.use('/', indexRouter);
app.use('/api/customers/v1', customersRouter);
 
///// app.use('/api/v1/customers', customersRouter);
 app.use('/api/dashboard/v1', uiRouter);
 
 
module.exports = app;
